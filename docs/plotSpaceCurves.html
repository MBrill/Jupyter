def helix(radius=1.0, h=2.0, left=0.0, right=1.0, num=200):
    """
    Erzeugung einer Polyline fr die Darstellung der Helix

    Parameters
    ----------
    radius : float, optional
        Radius der Helix. Default ist 1.0.
    h : float, optional
        Ganghhe. Default ist 2.0.
    left : float, optional
        Linke Grenze des Parameterintervalls. Default ist 0.0.
    right : float, optional
        Rechte Grenze des Parameterintervalls. Default ist 1.0.
    num : int, optional
        Anzahl der Punkte im Polygonzug. Default ist 200.

    Returns
    -------
    x : ndarray
        x Koordinaten.
    y : ndarray
        x Koordinaten.
    z : ndarray
        x Koordinaten.
    """
    t = np.linspace(left, right, num)
    x = radius * np.cos(2.0 * np.pi * t)
    y = radius * np.sin(2.0 * np.pi * t)
    z = h * t
    return x, y, z

def logSpiralHelix(a=2.0, b=0.08, h=1.0, left=0.0, right=8.0*np.pi, n=200):
    """
    Parametrisierung der Schraubenlinie ber einer logarithmischen Spirale.

    Parameters
    ----------
    a : float
        Parameter a der logarithmischen Spirale. Default ist 1.0.
    b : float
        Parameter b der logarithmischen Spirale. Default ist 0.08.
    h : float
        Ganghhe. Default ist 1.0.
    left : float
        Linke Grenze des Parameterintervalls. Default ist 0.0.
    right:  float
        Linke Grenze des Parameterintervalls. Default ist 6 pi.
    num: int
        Anzahl der Punkte fr die Polyline. Default ist 200.

    Returns
    -------
    x : float
        x-Koordinate.
    y : float
        y-Koordinate.
    z : float
        z-Koordinate.
    """
    t = np.linspace(left, right, n)
    r = a * np.exp(b*t)
    x = r * np.cos(t)
    y = r * np.sin(t)
    z = h * t
    return x, y, z

Darstellung mit einem Zylindermantel¶

Die Helix ist eine Schraubenlinie, die auf einem Zylindermantel abluft. In Plotly knnen wir geometrische Krper als Objekte vom Typ parametrized surface ausgeben. Dazu berechnen wir in der Funktion cylinder ein Dreiecksnetz eines Zylinders und ergeben das anschlieend an Plotly. Fr den Boden und den Deckel erzeugen implementieren wir die Funktino disk, die ein Dreiecks-Netz fr diese Kreisscheiben berechnet.

def cylinder(r=1.0, h=2.0, nu=100, nv=5):
    """
    Dreiecks-Netz fr einen Zylinder mit Radius r und Hhe 2. Die Auflsung der Kreise ist durch nu gegeben, 
    die Anzahl der Schichten in der Hhe durch den Parameter nv.
    
    Returns
    -------
    x : ndarray
        x Koordinaten.
    y : ndarray
        x Koordinaten.
    z : ndarray
        z Koordinaten.
    """
    theta = np.linspace(0, 2*np.pi, nu)
    v = np.linspace(0.0, h, nv )
    theta, v = np.meshgrid(theta, v)
    x = r*np.cos(theta)
    y = r*np.sin(theta)
    z = v
    return x, y, z


def circle_in_space(r=1.0, height=0.0, nt=100):
    """
    Eine grafische Darstellung eines Kreises mit Radius r. Die Kreisscheibe ist parallel zur xy-Ebene
    bei z=height.
    
    r - Radius des Kreises
    h - z-Hhe des Kreises
    nt - Anzahl der Punkte fr die grafische Ausgae

    Returns
    -------
    x : ndarray
        x Koordinaten.
    y : ndarray
        x Koordinaten.
    z : ndarray
        z Koordinaten.
    """
    theta = np.linspace(0, 2*np.pi, nt)
    x= r*np.cos(theta)
    y = r*np.sin(theta)
    z = np.full_like(theta, fill_value = height)
    return x, y, z
